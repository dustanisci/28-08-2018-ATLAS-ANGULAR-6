<b>Tecnologias utilizadas:</b>
<ul>
  <li>Angular 9</li>
  <li>Material</li>
  <li>HTML 5 </li>
  <li>SCSS / CSS 3</li>
  <li>TypeScript</li>
  <li>JavaScript</li>
</ul>
